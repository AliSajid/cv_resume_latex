\cventry{2020--Present}{\textbf{KINNET}}{Kinase INteraction NETwork modeling system}{R Package}{\url{https://github.com/CogDisResLab/KINNET}}{
    \begin{itemize}
        \item Implements a novel Bayesian method for inferring kinase interaction networks from PamGene PamChip activity data.
        \item Based on master's thesis work; under active development with CI/CD and test coverage.
        \item Provides upstream kinase assignment and peptide dependency modeling.
        \item GPL-3.0 license; hosted on GitHub and r-universe.
    \end{itemize}
}

\cventry{2020--Present}{\textbf{drugfindR}}{High-throughput drug repurposing toolkit}{R Package}{\url{https://github.com/CogDisResLab/drugfindR}}{
    \begin{itemize}
        \item Automates drug repurposing analysis using the iLINCS database and user-provided gene signatures.
        \item Calculates concordance values using perturbation signatures.
        \item Submitted to BioConductor; available via r-universe with 38 tagged releases.
        \item Comprehensive documentation and full test suite.
    \end{itemize}
}

\cventry{2021--Present}{\textbf{BioPathNet}}{Bioinformatics Workflow Framework}{R Package}{\url{https://github.com/CogDisResLab/BioPathNet}}{
    \begin{itemize}
        \item Provides foundational infrastructure for the 3PodReport system.
        \item Integrates GSEA, Enrichr, and LINCS-based signature analysis.
        \item Enables reproducible multi-omics analysis pipelines.
        \item Open-source on GitHub and r-universe.
    \end{itemize}
}

\cventry{2021--Present}{\textbf{3PodReport}}{Automated Multi-Omics Analysis Report Generator}{R Markdown Framework}{\url{https://github.com/CogDisResLab/3PodReport}}{
    \begin{itemize}
        \item Generates comprehensive reports integrating drugfindR, BioPathNet, and enrichment results.
        \item Used across multiple projects in the Cognitive Disorders Research Lab.
        \item Extensible and documented; maintained as a core lab reporting tool.
    \end{itemize}
}

\cventry{2021--Present}{\textbf{aaprop}}{Amino Acid Properties REST API}{Rust Application}{\url{https://github.com/Imamiland/aaprop}}{
    \begin{itemize}
        \item High-performance REST API for physicochemical amino acid property data.
        \item Built with actix-web in Rust; supports Shuttle and Docker deployments.
        \item Includes CLI and REST interface; 26 releases under Apache 2.0/MIT license.
        \item Live demo: \url{https://aaprop-c1io.shuttle.app/}
    \end{itemize}
}

\cventry{2022--Present}{\textbf{Variant Forge}}{Protein Sequence Comparison Service}{Web Application \& CLI}{\url{https://github.com/Imamiland/variant-forge}}{
    \begin{itemize}
        \item Calculates physicochemical distances between protein or DNA-translated amino acid sequences.
        \item Under active development for improvements and web application frontend.
        \item Provides both web-based interface and command-line tool.
        \item Available as a REST API service and Docker image under open-source license.
    \end{itemize}
}

\cventry{2021--Present}{\textbf{KRSA}}{Kinome Random Sampling Analyzer}{R Package}{\url{https://github.com/CogDisResLab/KRSA}}{
    \begin{itemize}
        \item Statistical analysis platform for PamChip array data used in kinase activity profiling.
        \item Maintained and improved since 2021; supports multiple published studies.
        \item Implements advanced statistical methods for differential kinase activity analysis.
    \end{itemize}
}

\cventry{2021--Present}{\textbf{Kaleidoscope}}{Gene Exploration and Expression Comparison Tool}{Web Application}{\url{https://github.com/CogDisResLab/Kaleidoscope}}{
    \begin{itemize}
        \item Aggregates gene-level expression data from curated RNA-seq and microarray datasets.
        \item Enables comparative analysis across diverse studies from a unified interface.
        \item Project lead and primary maintainer; currently in maintenance mode.
    \end{itemize}
}

\cventry{2020--Present}{\textbf{creedenzymatic}}{Comparative Upstream Kinase Analysis Toolkit}{R Package}{\url{https://github.com/CogDisResLab/creedenzymatic}}{
    \begin{itemize}
        \item Enables comparative analysis of upstream kinase inference methods: UKA, KRSA, PTM-SEA, and KEA3.
        \item Supports benchmarking and method validation in phosphoproteomic contexts.
    \end{itemize}
}
